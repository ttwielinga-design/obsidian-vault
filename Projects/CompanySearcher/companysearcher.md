    "import time\n",
    "from urllib.error import HTTPError\n",
    "import investpy\n",
    "import yfinance as yf\n",
    "import pandas as pd\n",
    "from tqdm import tqdm\n",
    "\n",
    "# 1) Define target countries/regions\n",
    "countries = [\n",
    "    'united states', 'canada',                          # North America\n",
    "    'united kingdom', 'germany', 'france', 'switzerland', 'netherlands',  # Europe\n",
    "    'japan', 'hong kong', 'australia', 'singapore'       # Asia-Pacific hubs\n",
    "]\n",
    "\n",
    "# 2) Suffix mapping for Yahoo Finance\n",
    "suffix_map = {\n",
    "    'United States': '',\n",
    "    'Canada': '.TO',\n",
    "    'United Kingdom': '.L',\n",
    "    'Germany': '.DE',\n",
    "    'France': '.PA',\n",
    "    'Switzerland': '.SW',\n",
    "    'Netherlands': '.AS',\n",
    "    'Japan': '.T',\n",
    "    'Hong Kong': '.HK',\n",
    "    'Australia': '.AX',\n",
    "    'Singapore': '.SI'\n",
    "}\n",
    "\n",
    "# 3) Build universe of (symbol, country)\n",
    "universe = []\n",
    "for country in countries:\n",
    "    try:\n",
    "        symbols = investpy.get_stocks_list(country=country)\n",
    "        universe += [(sym, country) for sym in symbols]\n",
    "        print(f\"Fetched {len(symbols)} tickers for {country.title()}\")\n",
    "    except Exception as e:\n",
    "        print(f\"Warning: could not fetch for {country.title()}: {e}\")\n",
    "\n",
    "# Deduplicate\n",
    "universe = list(set(universe))\n",
    "total = len(universe)\n",
    "print(f\"\\nTotal unique tickers in universe: {total}\\n\")\n",
    "\n",
    "# 4) Thresholds\n",
    "MIN_REV_CAGR    = 0.10\n",
    "MIN_EPS_CAGR    = 0.10\n",
    "MIN_ROOCE       = 0.15\n",
    "MIN_OP_MARGIN   = 0.20\n",
    "MIN_FCF_MARGIN  = 0.10\n",
    "MAX_DEBT_EBITDA = 3.0\n",
    "MIN_CUR_RATIO   = 1.5\n",
    "MIN_QUICK       = 1.0\n",
    "MIN_ROE         = 0.15\n",
    "MIN_ASSET_TO    = 1.0\n",
    "\n",
    "# 5) Helper to compute CAGR\n",
    "def cagr(start, end, n):\n",
    "    return (end / start) ** (1/n) - 1\n",
    "\n",
    "results = []\n",
    "\n",
    "# 6) Iterate with progress bar and compute metrics\n",
    "for symbol, country in tqdm(universe, desc=\"Screening companies\", unit=\"ticker\"):\n",
    "    # Build Yahoo symbol\n",
    "    yf_symbol = symbol + suffix_map.get(country.title(), '')\n",
    "\n",
    "    # Fetch data, skip on errors\n",
    "    try:\n",
    "        tk    = yf.Ticker(yf_symbol)\n",
    "        info  = tk.info\n",
    "        fin   = tk.financials\n",
    "        bs    = tk.balance_sheet\n",
    "        cf    = tk.cashflow\n",
    "    except (HTTPError, KeyError, ValueError, Exception):\n",
    "        continue\n",
    "\n",
    "    try:\n",
    "        # Ensure >3 years of history\n",
    "        rev = fin.loc['Total Revenue'].sort_index()\n",
    "        if rev.shape[0] < 4:\n",
    "            continue\n",
    "\n",
    "        # 3Y Revenue CAGR\n",
    "        rev_cagr = cagr(rev.iloc[-4], rev.iloc[-1], 3)\n",
    "\n",
    "        # 3Y EPS CAGR (using trailingEps proxy)\n",
    "        eps_now  = info.get('trailingEps')\n",
    "        eps_past = info.get('epsTrailingTwelveMonths')\n",
    "        eps_cagr = cagr(eps_past, eps_now, 3) if eps_now and eps_past and eps_past > 0 else None\n",
    "\n",
    "        # ROOCE\n",
    "        ebit = fin.loc['Ebit'].iloc[0]\n",
    "        nwc  = bs.loc['Total Current Assets'].iloc[0] - bs.loc['Total Current Liabilities'].iloc[0]\n",
    "        ppe  = bs.loc['Property Plant Equipment'].iloc[0]\n",
    "        rooce = ebit / (nwc + ppe) if (nwc + ppe) else None\n",
    "\n",
    "        # Operating margin\n",
    "        op_margin = ebit / rev.iloc[-1] if rev.iloc[-1] else None\n",
    "\n",
    "        # Free Cash Flow margin\n",
    "        op_cf  = cf.loc['Total Cash From Operating Activities'].iloc[0]\n",
    "        capex  = cf.loc['Capital Expenditures'].iloc[0]\n",
    "        fcf_margin = (op_cf - capex) / rev.iloc[-1] if rev.iloc[-1] else None\n",
    "\n",
    "        # Debt/EBITDA\n",
    "        debt   = bs.get('Long Term Debt', pd.Series([0])).iloc[0] + bs.get('Short Long Term Debt', pd.Series([0])).iloc[0]\n",
    "        ebitda = fin.get('Ebitda', pd.Series([None])).iloc[0]\n",
    "        debt_ebitda = debt / ebitda if ebitda else None\n",
    "\n",
    "        # Current & Quick ratios\n",
    "        ca    = bs.loc['Total Current Assets'].iloc[0]\n",
    "        cl    = bs.loc['Total Current Liabilities'].iloc[0]\n",
    "        inv   = bs.get('Inventory', pd.Series([0])).iloc[0]\n",
    "        cur_ratio  = ca / cl if cl else None\n",
    "        quick_ratio= (ca - inv) / cl if cl else None\n",
    "\n",
    "        # ROE\n",
    "        ni      = fin.loc['Net Income'].iloc[0]\n",
    "        eq_now  = bs.loc['Total Stockholder Equity'].iloc[0]\n",
    "        eq_prev = bs.loc['Total Stockholder Equity'].iloc[1] if bs.shape[1] > 1 else eq_now\n",
    "        roe     = ni / ((eq_now + eq_prev) / 2) if (eq_now + eq_prev) else None\n",
    "\n",
    "        # Asset turnover\n",
    "        total_assets = bs.loc['Total Assets'].iloc[0]\n",
    "        asset_to     = rev.iloc[-1] / total_assets if total_assets else None\n",
    "\n",
    "    except Exception:\n",
    "        continue\n",
    "\n",
    "    # 7) Check completeness & thresholds\n",
    "    metrics    = [rev_cagr, eps_cagr, rooce, op_margin, fcf_margin,\n",
    "                  debt_ebitda, cur_ratio, quick_ratio, roe, asset_to]\n",
    "    thresholds = [\n",
    "        rev_cagr    >= MIN_REV_CAGR,\n",
    "        eps_cagr    >= MIN_EPS_CAGR,\n",
    "        rooce       >= MIN_ROOCE,\n",
    "        op_margin   >= MIN_OP_MARGIN,\n",
    "        fcf_margin  >= MIN_FCF_MARGIN,\n",
    "        debt_ebitda <= MAX_DEBT_EBITDA,\n",
    "        cur_ratio   >= MIN_CUR_RATIO,\n",
    "        quick_ratio >= MIN_QUICK,\n",
    "        roe         >= MIN_ROE,\n",
    "        asset_to    >= MIN_ASSET_TO\n",
    "    ]\n",
    "\n",
    "    if all(m is not None for m in metrics) and all(thresholds):\n",
    "        results.append({\n",
    "            'Ticker': yf_symbol,\n",
    "            'Rev CAGR (3Y)': rev_cagr,\n",
    "            'EPS CAGR (3Y)': eps_cagr,\n",
    "            'ROOCE': rooce,\n",
    "            'Op Margin': op_margin,\n",
    "            'FCF Margin': fcf_margin,\n",
    "            'Debt/EBITDA': debt_ebitda,\n",
    "            'Current Ratio': cur_ratio,\n",
    "            'Quick Ratio': quick_ratio,\n",
    "            'ROE': roe,\n",
    "            'Asset Turnover': asset_to\n",
    "        })\n",
    "    time.sleep(0.3)\n",
    "\n",
    "# 8) Output results\n",
    "print(f\"\\n\\nScreen complete. Found {len(results)} companies passing all criteria.\\n\")\n",
    "df = pd.DataFrame(results)\n",
    "print(df.sort_values('Rev CAGR (3Y)', ascending=False).reset_index(drop=True))\n"
   ]
